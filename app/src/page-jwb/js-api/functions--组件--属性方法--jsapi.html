类　型：Object
解　释：返回值有2个属性
1.width 浏览器包括滚动条滚动的距离总和宽度 类型：Number
2.height 浏览器包括滚动条滚动的距离总和高度 类型：Number
说　明： 获取浏览器窗口的宽和高，宽度高度都包括浏览器向左或者向右滚动的距离

		/**
		 * [getViewPortSize 获取浏览器窗口的宽和高]
		 * @return {[json]} [宽和高]
		 */
		function getViewPortSize(){
		    var winWidth = document.documentElement.clientWidth || document.body.clientWidth;
		    var winHeight = document.documentElement.clientHeight || document.body.clientHeight;
		    var scrollL = document.documentElement.scrollLeft || document.body.scrollLeft;
		    var scrollT = document.documentElement.scrollTop || document.body.scrollTop;
		    return {"width": winWidth + scrollL, "height":winHeight + scrollT};
		}

返回值：
类　型：String

		/**
		 * [checkUqiString 检查过滤特殊字符串]
		 * @param  {[string]} s [待处理的字符串]
		 * @return {[string]}   [过滤后的字符串]
		 */
		function checkUqiString(s) {
		    var pattern = new RegExp("[`~!@#$^&*()=|{}':;',\\[\\].<>/?~！@#￥……&*（）&;—|{}【】‘；：”“'\"。，、？%+_-]","i");
		    var rs = "";
		    s = s.replace(/(\\|\s)/g, "");
		    s = s.substring(0, 100);
		    for (var i = 0; i < s.length; i++) {
		        rs = rs + s.substr(i, 1).replace(pattern, '');
		    }
		    return rs;
		}

		function dateCompare(timeStart, timeEnd){
		    var aStart = timeStart.match(/\d+/g);
		    var aEnd = timeEnd.match(/\d+/g);
		    var iTimeStart = new Date(aStart[0],aStart[1],aStart[2]).getTime();
		    var iTimeEnd = new Date(aEnd[0],aEnd[1],aEnd[2]).getTime();

		    if(iTimeStart < iTimeEnd){
		        return timeStart;
		    }else if(iTimeStart > iTimeEnd){
		        return timeEnd;
		    }
		}
getSmallScreenSize(iTar, iDis, sTop) - layer弹窗hack小屏幕高度
参数一：
类　型：Number
解　释：要设置的弹窗的高度值，根据UI设计图量出的高度 默认：无
参数二：
类　型：Number
解　释：屏幕较小时上下的留白 默认：无

类　型：String
解　释：页面结构头部区域的class名，无需带 . 号 默认：header_top
返回值：
类　型：Object
解　释：返回合适的高度值和合适的layer的offset属性对应的值
1.height 合适的layer的area的高度值 类型：String,如：100px
2.offset 合适的layer的offset属性对应的值 类型：String，如100px或者auto
说　明： layer弹窗hack小屏幕高度，有的时候根据UI设计图量出的高度过大，用这个方法可以hack小屏浏览器的情况
位　置： src/js/page/page_frame_zheng.js 
		/**
		 * [getSmallScreenSize layer弹窗hack小屏幕高度]
		 * @param  {[number]} iTar [要设置的弹窗的高度值]
		 * @param  {[number]} iDis [屏幕较小时上下的留白]
		 * @param  {[string]} sTop [骨架头部元素]
		 * @return {[object]}      [理想的高度值height，弹窗的上边界值offset]
		 */
		function getSmallScreenSize (iTar, iDis, sTop) {
		    var oTop = null;
		    var sHei = null;
		    var sOffset = null;
		    var iWinHei = parent.getViewPortSize().height;

		    // 获取弹窗的理想高度
		    sTop ? oTop = parent.$("." + sTop) : oTop = parent.$('.header_top') ;
		    iWinHei > iTar ? sHei = iTar + "px" : sHei = iWinHei - oTop.outerHeight() - iDis*2 + "px" ;
		    // 获取窗口上边界距离
		    iWinHei > iTar ? sOffset = "auto" : sOffset = oTop.outerHeight() + iDis + "px";

		    return {height: sHei, offset: sOffset};
		}

		/**
		 * [showBigImg 图片预览大图]
		 * @param  {[string]} obj  [小图]
		 * @param  {[string]} oImg [大图]
		 * @return {[type]}      [description]
		 */
		function showBigImg(obj, oImg){
		    $(obj).hover(function(ev){
		        var oEv = ev || event;
		        var iX = oEv.pageX;
		        var iY = oEv.pageY;
		        var sSrc = $(this).attr("src");

		        $(oImg).attr("src", sSrc);
		        $(oImg).css({
		            "left": iX,
		            "top": iY
		        }).stop(true, true).fadeIn(100);
		    },function(){
		        $(oImg).stop(true, true).fadeOut(100);
		    })
		}
